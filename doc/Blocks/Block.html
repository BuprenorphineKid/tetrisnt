def initialize(pic, win, x, y)

        @win = win
        @pic = pic

        @collides = false    
        
        @scale = (@win.height / @pic.height) / 8
        @x = x
        @y = y
        @state = "falling"
        @moveable = true

        @vel = 7
        @speed = 1
end
