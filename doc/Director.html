def determine scene
  case scene
  when /[Ll][Ee][Vv][Ee][Ll]|.*\d/
    level
  when /[TtBb][IiEe][TtGg][LlIi][EeNn]/
    title
  when /[SsPp][TtAa][AaUu][RrSs][TtEe]|/
    pause
  end
end
